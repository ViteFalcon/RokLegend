SET(RO_CORE_LIBRARY "RoCore")

MACRO(RO_TARGET_LINK_LIBRARIES TARGET MODULES)
    SET(MODULE_LIBS "")
    FOREACH(LIB_NAME ${MODULES})
        LIST(APPEND MODULE_LIBS debug ${ROKLEGEND_LIB_DIR_DBG}/${LIB_NAME}_d.lib optimized ${ROKLEGEND_LIB_DIR_REL}/${LIB_NAME}.lib)
    ENDFOREACH()
    TARGET_LINK_LIBRARIES(${TARGET} ${MODULE_LIBS})
ENDMACRO()

ADD_SUBDIRECTORY(core)
SET(RO_CORE_DIR ${ROKLEGEND_SOURCE_DIR}/core)

# Make sure to include RoCore only after creating RoCore project
INCLUDE_DIRECTORIES(${RO_CORE_DIR})

    ${RO_CORE_LIBRARY}
